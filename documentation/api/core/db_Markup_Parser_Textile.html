Prepare the parsing of a Textile string, the real parsing is done in {@link _parse()}

        'tag'        => '[tag="a" attr=val]',
        'type'       => Zend_Markup::TYPE_TAG,
        'name'       => 'tag',
        'stoppers'   => array('[/]', '[/tag]'),
        'attributes' => array(
            'tag'  => 'a',
            'attr' => 'val'
        )

    array(
        'tag'   => 'value',
