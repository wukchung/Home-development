Zend_Reflection Beispiele

Example #1 Durchführen von Reflection an einer Datei

  1. $r = new Zend_Reflection_File($filename);
  2. printf(
  3.     "===> Die Datei %s\n".
  4.     "     hat %d Zeilen\n",
  5.     $r->getFileName(),
  6.     $r->getEndLine()
  7. );
  8.  
  9. $classes = $r->getClasses();
  10. echo "     Sie hat " . count($classes) . ":\n";
  11. foreach ($classes as $class) {
  12.     echo "         " . $class->getName() . "\n";
  13. }
  14.  
  15. $functions = $r->getFunctions();
  16. echo "     Sie hat " . count($functions) . ":\n";
  17. foreach ($functions as $function) {
  18.     echo "         " . $function->getName() . "\n";
  19. }

Example #2 Durchführen von Reflection an einer Klasse

  1. $r = new Zend_Reflection_Class($class);
  2.  
  3. printf(
  4.     "Der Klassen-Level Docblock hat die Kurzbeschreibung: %s\n".
  5.     "Der Klassen-Level Docblock hat die Langbeschreibung:\n%s\n",
  6.     $r->getDocblock()->getShortDescription(),
  7.     $r->getDocblock()->getLongDescription(),
  8. );
  9.  
  10. // Die Deklarierte Datei Reflektion
  11. $file = $r->getDeclaringFile();

Example #3 Durchführen von Reflection an einer Methode

  1. $r = new Zend_Reflection_Method($class, $name);
  2.  
  3. printf(
  4. "Die Methode '%s' hat einen Rückgabetyp von %s",
  5.     $r->getName(),
  6.     $r->getReturn()
  7. );
  8.  
  9. foreach ($r->getParameters() as $key => $param) {
  10.     printf(
  11.         "Der Parameter an Position '%d' ist vom Typ '%s'\n",
  12.         $key,
  13.         $param->getType()
  14.     );
  15. }

Example #4 Durchführen von Reflection an einem Docblock

  1. $r = new Zend_Reflection_Method($class, $name);
  2. $docblock = $r->getDocblock();
  3.  
  4. printf(
  5.     "Die Kurzbeschreibung: %s\n".
  6.     "Die Langbeschreibung:\n%s\n",
  7.     $r->getDocblock()->getShortDescription(),
  8.     $r->getDocblock()->getLongDescription(),
  9. );
  10.  
  11. foreach ($docblock->getTags() as $tag) {
  12.     printf(
  13.         "Das Hinweis-Tag '%s' hat die Beschreibung '%s'\n",
  14.         $tag->getName(),
  15.         $tag->getDescription()
  16.     );
  17. }
